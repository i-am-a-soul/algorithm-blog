# 区间

[AcWing-362-区间](https://www.acwing.com/problem/content/364/)

## 分析

见《进阶指南》第`394`页。

## 实现

```cpp
#include <iostream>
#include <cstring>
#include <queue>
#define inf 0x3f3f3f3f
using namespace std;

const int N = 50010, offset = 1;
struct edge {
    int to, next, w;
};
edge e[3 * N];
int idx, head[N];
struct node {
    int idx, dis;
    bool operator < (const node& o) const {
        return dis > o.dis;
    }
};
int n;
int dis[N];
bool mark[N];

void add_edge (int u, int v, int w) {
    e[idx].w = w;
    e[idx].to = v;
    e[idx].next = head[u];
    head[u] = idx ++;
}
void spfa (int s) {
    memset(dis, 0x3f, sizeof(dis));
    dis[s] = 0;
    queue<int> q;
    q.push(s);
    mark[s] = true;
    while (!q.empty()) {
        int cur = q.front();
        q.pop();
        mark[cur] = false;
        for (int i = head[cur]; i != -1; i = e[i].next) {
            int to = e[i].to, w = e[i].w;
            if (dis[cur] + w < dis[to]) {
                dis[to] = dis[cur] + w;
                if (mark[to] == false) {
                    q.push(to);
                    mark[to] = true;
                }
            }
        }
    }
}

int main () {
    memset(head, -1, sizeof(head));
    cin >> n;
    int l = inf, r = -inf;
    for (int i = 1, a, b, c; i <= n; ++ i) {
        cin >> a >> b >> c;
        add_edge(b + offset, a - 1 + offset, -c);
        l = min(l, a);
        r = max(r, b);
    }
    for (int k = l; k <= r; ++ k) {
        add_edge(k + offset, k - 1 + offset, 0);
        add_edge(k - 1 + offset, k + offset, 1);
    }

    spfa(r + offset);
    printf("%d\n", -dis[l - 1 + offset]);
    return 0;
}
```
